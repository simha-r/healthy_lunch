def all
  p env["omniauth.auth"]
  user = User.from_omniauth(env["omniauth.auth"], current_user)
  session[:login_type]=user.login_type
  if user.persisted?
    user.confirm!
    sign_in_and_redirect(user)
  else
    session["devise.user_attributes"] = user.attributes
    redirect_to new_user_registration_url
  end
end
